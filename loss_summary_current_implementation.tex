\section{Goal}

This note summarizes all training losses in current \texttt{train.py}, grouped by role,
and states what is active for the current depth-time-resync run configuration.

\section{Grouped Losses (Current Codebase)}

\subsection*{Group A: Core diffusion objective}
\begin{itemize}[leftmargin=*]
  \item \textbf{\(L_{\text{gen}}\)}: main diffusion generation loss on backbone output.
\end{itemize}

\subsection*{Group B: Shortcut direct learning}
\begin{itemize}[leftmargin=*]
  \item \textbf{\(L_{\text{direct}}\)}: predictor learns direct transition between source/target states.
\end{itemize}

\subsection*{Group C: Shortcut bootstrap consistency (legacy)}
\begin{itemize}[leftmargin=*]
  \item \textbf{\(L_{\text{boot}}\)}: compositional consistency of shortcut transitions using EMA predictor as teacher.
\end{itemize}

\subsection*{Group D: Legacy output distillation}
\begin{itemize}[leftmargin=*]
  \item \textbf{\(L_{\text{output\_distill}}\)}: distill final output using resumed backbone from predicted intermediate hidden.
\end{itemize}

\subsection*{Group E: Depth-time-resync core}
\begin{itemize}[leftmargin=*]
  \item \textbf{\(L_{\text{depth}}\)}: layer transition at same time.
  \item \textbf{\(L_{\text{time}}\)}: time transition at same layer.
  \item \textbf{\(L_{\text{dt}}\)}: joint depth-time transition.
\end{itemize}

\subsection*{Group F: Depth-time-resync consistency}
\begin{itemize}[leftmargin=*]
  \item \textbf{\(L_{\text{depth\_cons}}\)}: composition consistency along depth transitions.
  \item \textbf{\(L_{\text{time\_cons}}\)}: composition consistency along time transitions.
  \item \textbf{\(L_{\text{dt\_cons}}\)}: composition consistency along depth-time transitions.
\end{itemize}

\subsection*{Group G: Depth-time output-time distillation}
\begin{itemize}[leftmargin=*]
  \item \textbf{\(L_{\text{out\_time}}\)}: from predicted depth-time hidden, resume tail and match output at target time.
\end{itemize}

\subsection*{Group H: Auxiliary regularization}
\begin{itemize}[leftmargin=*]
  \item \textbf{\(L_{\text{skip\_fm}}\)}: skip-in-loop feature matching regularizer.
  \item \textbf{\(L_{\text{private}}\)}: common/private activation regularization.
\end{itemize}
